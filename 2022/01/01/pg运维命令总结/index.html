1 psql客户端工具

  • help –总的帮助 
  • \h –SQL commands级的帮助 
  • ? –gsql commands级的帮助 \dn[S+]，其中[PATTERN]表示可以使用正则表达式，比如\dns f*表示列出所有f开头的schema 
  • \x 按列显示，类似mysql的\G 
  • \l 查看数据库
  • \l+查看数据库及大小
  • \c 切换数据库
  • \d 查看表
  • \d 表名 查看表的定义
  • \d 索引 查看索引信息
  • \d 表名?或索引名?通配符
  • \d+ 表名 显示比\d更详细的信息，例如列的注释、表中出现的OID等
  • \dt 只显示匹配表
  • \di 只显示匹配索引
  • \ds 只显示匹配序列
  • \dv 只显示匹配视图
  • \df 只显示匹配函数
  • \dD查询域
  • \dn 列出所有Schema
  • \db 列出所有表空间
  • \du或\dg 列出所有用户或角色
  • \dp或\z 表名用于显示表的权限分配情况
  • \e 打开文本编辑器。
  • \conninfo 列出当前数据库和连接的信息。
  • gsql -E 或者\set ECHO_HIDDEN on显示斜杠\后面的命令对应的语句，ECHO_HIDDEN必须大写

    4.4 表

  • 查看表名 
select * from pg_tables WHERE tablename NOT LIKE 'pg%' ORDER BY tablename;--普通用户也能查询到整个数据库下的所有表,标准语句

select table_catalog,table_schema,table_name,table_type,is_insertable_into,is_typed from information_schema.tables where table_type = 'BASE TABLE';--superuser可以查询到所有表，普通用户只能查询到自己owner的表，但是查询结果居然有视图在里面，所以不是标准语句
  • 按占空间大小，顺序查看所有表的大小
    select schemaname,relname, pg_size_pretty(pg_total_relation_size(relid)) from pg_stat_user_tables order by pg_relation_size(relid) desc;--包含表和索引大小

    select schemaname,relname, pg_size_pretty(pg_table_size(relid)) from pg_stat_user_tables order by pg_relation_size(relid) desc;--不包括索引（但包括TOAST、空闲空间映射fsm和可见性映射）
  • 获取当前库表和索引大小
    SELECT *,
  Pg_size_pretty(total_bytes) AS total,
  Pg_size_pretty(index_bytes) AS INDEX,
  Pg_size_pretty(toast_bytes) AS toast,
  Pg_size_pretty(table_bytes) AS TABLE
FROM (SELECT *,
  total_bytes - index_bytes - Coalesce(toast_bytes, 0) AS table_bytes
  FROM(SELECT c.oid,
              nspname                                     AS table_schema,
              relname                                     AS TABLE_NAME,
              c.reltuples                                 AS row_estimate,
              Pg_total_relation_size(c.oid)                 AS total_bytes,
              Pg_indexes_size(c.oid)                         AS index_bytes,
              Pg_total_relation_size(reltoastrelid)       AS toast_bytes
              FROM pg_class c
              LEFT JOIN pg_namespace n
              ON n.oid = c.relnamespace
              WHERE relkind = 'r') a
WHERE table_schema = 'PUBLIC'
ORDER BY total_bytes DESC) a;
  • 查看超过1GB的表或者索引大小
    ```sql
  • -表
    select schemaname,relname,size from(
    select split_part(pg_ls_dir(relation_path),’.’,1) as relid,count(1)-1 as size from (
    select replace(PG_RELATION_FILEPATH(relid),reverse(split_part(reverse(PG_RELATION_FILEPATH(relid)),’/‘,1)),’’) as relation_path
    from pg_stat_user_tables group by 1
    )z
    group by 1
    having count(1) > 1
    )x,pg_stat_user_tables y
    where x.relid::oid = pg_relation_filenode(y.relid)
    order by size desc;

select relname,
case relkind
when ‘r’ then ‘table’
when ‘i’ then ‘index’
when ‘t’ then ‘toast’
end as relkind,
size||’GB+’ as size from(
select split_part(pg_ls_dir(relation_path),’.’,1) as relid,count(1)-1 as size from (
select replace(PG_RELATION_FILEPATH(relid),reverse(split_part(reverse(PG_RELATION_FILEPATH(relid)),’/‘,1)),’’) as relation_path
from pg_stat_user_tables group by 1
)z
group by 1
having count(1) > 1
)x,pg_class y
where x.relid::oid = y.relfilenode

  • 查询超过1GB的索引
select schemaname,indexrelname,relname,size from(
select split_part(pg_ls_dir(relation_path),'.',1) as relid,count(1)-1 as size from (
select replace(PG_RELATION_FILEPATH(relid),reverse(split_part(reverse(PG_RELATION_FILEPATH(relid)),'/',1)),'') as relation_path
from PG_STAT_USER_INDEXES group by 1
)z
group by 1
having count(1) > 1
)x,PG_STAT_USER_INDEXES y
where x.relid::oid = pg_relation_filenode(y.relid)
order by size desc;

    4.11 类型

  • 查看自定义类型ID的定义：
    select attname, format_type(atttypid, atttypmod)
from pg_type
     join pg_class on pg_class.oid = pg_type.typrelid
     join pg_attribute on pg_attribute.attrelid = pg_class.oid
where typname = 'ID'
order by attnum;
  • 查看域US_POSTAL_CODE的定义
    select n.nspname as "schema",
           t.typname as "name",
           pg_catalog.format_type(t.typbasetype, t.typtypmod) as "type",
           trim(leading
           coalesce((select 'collate' || c.collname from pg_catalog.pg_collation c, pg_catalog.pg_type bt
where c.oid = t.typcollation and bt.oid = t.typbasetype and t.typcollation <> bt.typcollation), '') ||
case when t.typnotnull then ' not null' else '' end ||
case when t.typdefault is not null then ' default ' || t.typdefault else '' end
) as "modifier",
pg_catalog.array_to_string(array(
select pg_catalog.pg_get_constraintdef(r.oid, true) from pg_catalog.pg_constraint r where t.oid = r.contypid
), ' ') as "check"
from pg_catalog.pg_type t
left join pg_catalog.pg_namespace n on n.oid = t.typnamespace
where t.typtype = 'd'
and t.typname operator(pg_catalog.~) '^(us_postal_code)$'
and pg_catalog.pg_type_is_visible(t.oid)
order by 1, 2;

6.2 查看锁

select a.locktype,a.database,a.pid,a.mode,a.relation,b.relname from pg_locks a join pg_class b on a.relation = b.oid;

SELECT locker.pid,
                pc.relname,
                locker.mode,
                locker_act.application_name,
                least(query_start,xact_start) start_time,
                locker_act.state,
                CASE granted
                WHEN 'f' THEN 'wait_lock'
                WHEN 't' THEN 'get_lock'
                END lock_satus,
                current_timestamp - least(query_start,xact_start) AS runtime,
                locker_act.query
FROM pg_locks locker,pg_stat_activity locker_act, pg_class pc
WHERE locker.pid=locker_act.pid
AND NOT locker.pid=pg_backend_pid()
AND application_name<>'pg_statsinfod'
AND locker.relation = pc.oid
AND pc.reltype<>0
AND pc.relname='表'
ORDER BY runtime desc;
