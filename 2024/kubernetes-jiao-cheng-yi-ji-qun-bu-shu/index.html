Kubernetes教程（一）集群部署

容器
Kubernetes
发布日期:   2024-06-07

1. 环境准备

1.1 系统配置

# 设置主机名
echo 'master' > /etc/hostname  # 在master节点执行
echo 'node01' > /etc/hostname  # 在node01节点执行

# 配置hosts文件
cat >> /etc/hosts <<EOF
192.168.137.226 master
192.168.137.227 node01
EOF

# 关闭防火墙
systemctl stop firewalld
systemctl disable firewalld

1.2 禁用swap

swapoff -a
sed -i '/\<swap\>/ s/^/#/' /etc/fstab

2. 安装Docker

# 安装指定版本Docker
yum -y install docker-ce-19.03.13-3.el8

# 配置Docker
tee > /etc/docker/daemon.json <<EOF
{
  "exec-opts": ["native.cgroupdriver=systemd"],
  "log-driver": "json-file",
  "log-opts": {
    "max-size": "100m"
  },
  "storage-driver": "overlay2",
  "storage-opts": [
    "overlay2.override_kernel_check=true"
  ],
  "registry-mirrors": ["https://docker.registry.cyou",
    "https://docker-cf.registry.cyou",
    "https://dockercf.jsdelivr.fyi",
    "https://docker.jsdelivr.fyi",
    "https://dockertest.jsdelivr.fyi",
    "https://mirror.aliyuncs.com",
    "https://dockerproxy.com",
    "https://mirror.baidubce.com",
    "https://docker.m.daocloud.io",
    "https://docker.nju.edu.cn",
    "https://docker.mirrors.sjtug.sjtu.edu.cn",
    "https://docker.mirrors.ustc.edu.cn",
    "https://mirror.iscas.ac.cn",
    "https://docker.rainbond.cc"]
}
EOF

# 启动Docker
systemctl enable docker
systemctl start docker

useradd -m jinli
# 将用户加入docker组
usermod -a -G docker jinli

3. 网络初始化配置

cat > /usr/bin/k8s-init.sh <<EOF
#!/bin/bash
iptables -P FORWARD ACCEPT
EOF

chmod +x /usr/bin/k8s-init.sh

cat > /usr/lib/systemd/system/k8s-init.service <<EOF
[Unit]
Description=Kubernetes Init Task
After=multi-user.target

[Service]
ExecStart=/usr/bin/k8s-init.sh

[Install]
WantedBy=multi-user.target
EOF

systemctl enable k8s-init
systemctl start k8s-init

4. 安装Kubernetes组件

4.1 配置yum源

cat > /etc/yum.repos.d/kubernetes.repo <<EOF
[kubernetes]
name=Kubernetes
baseurl=https://mirrors.aliyuncs.com/kubernetes/yum/repos/kubernetes-el7-x86_64/
enabled=1
gpgcheck=1
gpgkey=https://mirrors.aliyuncs.com/kubernetes/yum/doc/yum-key.gpg
       https://mirrors.aliyuncs.com/kubernetes/yum/doc/rpm-package-key.gpg
repo_gpgcheck=1
EOF

yum clean all
yum makecache

4.2 安装kubeadm, kubelet和kubectl

yum install -y kubelet-1.20.1-0 kubeadm-1.20.1-0 kubectl-1.20.1-0
systemctl enable kubelet

5. 初始化Master节点

# 初始化集群
kubeadm init \
  --kubernetes-version=1.20.1 \
  --image-repository registry.aliyuncs.com/google_containers \
  --pod-network-cidr=192.168.2.0/24 \
  --service-cidr=10.96.0.0/12

# 配置kubectl
mkdir -p $HOME/.kube
sudo cp -i /etc/kubernetes/admin.conf $HOME/.kube/config
sudo chown $(id -u):$(id -g) $HOME/.kube/config

6. 安装网络插件(Calico)

mkdir -p ~/k8s/calico
cd ~/k8s/calico
curl -L https://docs.projectcalico.org/v3.18/manifests/calico.yaml -o calico-v3.18.yaml
sed -i -e "s?192.168.0.0/16?192.168.2.0/24?g" calico-v3.18.yaml
kubectl apply -f calico-v3.18.yaml

7. 加入Worker节点

在master节点获取加入命令：

kubeadm token create --print-join-command

在node01节点执行：

kubeadm reset
kubeadm join 192.168.137.226:6443 --token <token> --discovery-token-ca-cert-hash sha256:<hash>

8. 配置kube-proxy使用ipvs模式

kubectl edit cm kube-proxy -n kube-system
# 修改mode为ipvs

9. 验证集群状态

kubectl get nodes
kubectl get pods -n kube-system

3. 常见问题排查

3.1 证书错误

# 如果出现x509证书错误
rm -rf $HOME/.kube
mkdir -p $HOME/.kube
sudo cp -i /etc/kubernetes/admin.conf $HOME/.kube/config
sudo chown $(id -u):$(id -g) $HOME/.kube/config

3.2 移除节点

# 优雅移除节点
kubectl drain node01 --ignore-daemonsets --delete-emptydir-data
kubectl delete node node01

# 在节点上清理
kubeadm reset

总结

本教程涵盖了从Kubernetes集群部署的完整流程，包括：

  1. 环境准备和依赖安装
  2. Kubernetes集群初始化
  3. 网络插件安装
  4. 工作节点加入
  5. 常见问题排查方法
